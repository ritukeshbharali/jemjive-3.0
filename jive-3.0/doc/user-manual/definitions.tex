%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% The Figure environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


% Environment for displaying a (framed) figure

\newlength{\FigureWidth}
\setlength{\FigureWidth}{\textwidth}
\addtolength{\FigureWidth}{-1cm}

\newenvironment{Figure}
{
  \begin{figure}
    \begin{center}
      \begin{minipage}{\FigureWidth}
}
{
      \end{minipage}
    \end{center}
    \vspace{0mm}
  \end{figure}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% The Table environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


% Environment for displaying a (framed) table

\newenvironment{Table}
{
  \begin{table}
    \begin{center}
      \begin{minipage}{\FigureWidth}
}
{
      \end{minipage}
    \end{center}
    \vspace{2mm}
  \end{table}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% The SideStep environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\newlength{\SideStepWidth}
\setlength{\SideStepWidth}{\textwidth}
\addtolength{\SideStepWidth}{-1cm}

\newenvironment{SideStep}[1]
{
  \begin{figure}[t!]
    \begin{center}
      \begin{tabular}{|p{\SideStepWidth}|}
         \hline
         \begin{center}
            {\large \textbf{#1}}
         \end{center}
         \small
         \setlength{\parindent}{1em}
         \BlankLine
}
{
         \\ \\ \hline
      \end{tabular}
    \end{center}
  \end{figure}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% The Algorithm environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newlength{\AlgorithmWidth}
\setlength{\AlgorithmWidth}{\FigureWidth}
\addtolength{\AlgorithmWidth}{-2cm}

% Environment for typesetting algorithms. 

\newenvironment{Algorithm}[1][\AlgorithmWidth]
{ 
  \begin{center}
    \begin{tabular}{|p{#1}|}
      \hline
      \begin{tabbing}
        12\=12\=12\=12\=12\=12\=12\=12\=12\=12\=12\= \+ \kill
} 
{ 
      \end{tabbing}
      \\ \hline
    \end{tabular}
  \end{center}
}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% The ShortAlgorithm environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


% Environment for typesetting short algorithms, without a caption

\newenvironment{ShortAlgorithm}
{
  \begin{list}{}{
    \setlength{\leftmargin}{1cm}
    \setlength{\rightmargin}{1cm}
    \setlength{\labelwidth}{0cm}
    \setlength{\itemindent}{0cm}
  }
  
  \item
  \begin{tabbing}
    12\=12\=12\=12\=12\=12\=12\=12\=12\=12\=12\= \kill
}
{
  \end{tabbing}
  \end{list}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% The Remarks environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


\newcounter{RemarksCounter}

% Environment for creating a list of remarks. Arguments: none 

\newenvironment{Remarks}
{
  \begin{list}
    {\textbf{Remark \arabic{RemarksCounter}:}}
    {\usecounter{RemarksCounter}
    \settowidth{\labelwidth}{\textbf{Remark 99}}
    \setlength{\leftmargin}{0.5cm}
    \setlength{\rightmargin}{\leftmargin}
    \setlength{\itemindent}{\labelwidth}
    \addtolength{\itemindent}{\labelsep}
    \addtolength{\itemindent}{-\leftmargin}}
} 
{ 
  \end{list} 
}
